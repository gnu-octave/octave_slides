\begin{column}{0.3\textwidth}
\textbf{classdef classes}
\begin{lstlisting}
classdef myclass
  properties
    value
  endproperties

  methods (Access = private)
    function a = secret (obj, ...)
      ...
    endfunction
  endmethods

  methods
    function obj = myclass ()
      obj.value = 42;
      ...
    endfunction

    function disp (obj)
      disp (obj.value)
    endfunction
  endmethods
endclassdef
\end{lstlisting}
